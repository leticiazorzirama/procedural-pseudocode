#Faça um algoritmo que leia um valor e escreva: 0, se o valor é zero; 1, se o valor é maior que zero; -1 - se o valor é negativo.

var

  x : inteiro

inicio

  leia(x) 

  se x = 0 entao
    escreva("0") 
  fimse

  se x > 0 entao
    escreva(1) 
  fimse

  se x < 0 entao
    escreva(-1) 
  fimse

fimalgoritmo
